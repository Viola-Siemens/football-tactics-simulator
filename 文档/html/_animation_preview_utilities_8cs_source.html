68  public static void PreviewFromCurves(GameObject animatorRoot, IEnumerable<EditorCurveBinding> keys)
69  {
70  if (!AnimationMode.InAnimationMode())
71  return;
72 
73  var avatarRoot = GetAvatarRoot(animatorRoot);
74  foreach (var binding in keys)
75  {
76  if (IsAvatarBinding(binding) || IsEuler(binding))
77  continue;
78 
79  bool isTransform = typeof(Transform).IsAssignableFrom(binding.type);
80  if (isTransform && binding.propertyName == AnimatorBindingCache.TRPlaceHolder)
81  AddTRBinding(animatorRoot, binding);
82  else if (isTransform && binding.propertyName == AnimatorBindingCache.ScalePlaceholder)
83  AddScaleBinding(animatorRoot, binding);
84  else
85  AnimationMode.AddEditorCurveBinding(avatarRoot, binding);
86  }

89  public static AnimationClip CreateDefaultClip(GameObject animatorRoot, IEnumerable<EditorCurveBinding> keys)
90  {
91  AnimationClip animClip = new AnimationClip() { name = "DefaultPose" };
92  var keyFrames = new[] {new Keyframe(0, 0)};
93  var curve = new AnimationCurve(keyFrames);
94  bool rootMotion = false;
95  var avatarRoot = GetAvatarRoot(animatorRoot);
96 
97  foreach (var binding in keys)
98  {
99  if (IsRootMotion(binding))
100  {
101  rootMotion = true;
102  continue;
103  }
104 
105  if (typeof(Transform).IsAssignableFrom(binding.type) && binding.propertyName == AnimatorBindingCache.TRPlaceHolder)
106  {
107  if (string.IsNullOrEmpty(binding.path))
108  rootMotion = true;
109  else
110  {
111  var transform = animatorRoot.transform.Find(binding.path);
112  if (transform != null)
113  {
114  var pos = transform.localPosition;
115  var rot = transform.localRotation;
116  animClip.SetCurve(binding.path, typeof(Transform), k_PosX, SetZeroKey(curve, keyFrames, pos.x));
117  animClip.SetCurve(binding.path, typeof(Transform), k_PosY, SetZeroKey(curve, keyFrames, pos.y));
118  animClip.SetCurve(binding.path, typeof(Transform), k_PosZ, SetZeroKey(curve, keyFrames, pos.z));
119  animClip.SetCurve(binding.path, typeof(Transform), k_RotX, SetZeroKey(curve, keyFrames, rot.x));
120  animClip.SetCurve(binding.path, typeof(Transform), k_RotY, SetZeroKey(curve, keyFrames, rot.y));
121  animClip.SetCurve(binding.path, typeof(Transform), k_RotZ, SetZeroKey(curve, keyFrames, rot.z));
122  animClip.SetCurve(binding.path, typeof(Transform), k_RotW, SetZeroKey(curve, keyFrames, rot.w));
123  }
124  }
125 
126  continue;
127  }
128 
129  if (typeof(Transform).IsAssignableFrom(binding.type) && binding.propertyName == AnimatorBindingCache.ScalePlaceholder)
130  {
131  var transform = animatorRoot.transform.Find(binding.path);
132  if (transform != null)
133  {
134  var scale = transform.localScale;
135  animClip.SetCurve(binding.path, typeof(Transform), k_ScaleX, SetZeroKey(curve, keyFrames, scale.x));
136  animClip.SetCurve(binding.path, typeof(Transform), k_ScaleY, SetZeroKey(curve, keyFrames, scale.y));
137  animClip.SetCurve(binding.path, typeof(Transform), k_ScaleZ, SetZeroKey(curve, keyFrames, scale.z));
138  }
139 
140  continue;
141  }
142 
143  // Not setting curves through AnimationUtility.SetEditorCurve to avoid reentrant
144  // onCurveWasModified calls in timeline. This means we don't get sprite curves
145  // in the default clip right now.
146  if (IsAvatarBinding(binding) || IsEulerHint(binding) || binding.isPPtrCurve)
147  continue;
148 
149  float floatValue;
150  AnimationUtility.GetFloatValue(avatarRoot, binding, out floatValue);
151  animClip.SetCurve(binding.path, binding.type, binding.propertyName, SetZeroKey(curve, keyFrames, floatValue));
152  }
153 
154  // add root motion explicitly.
155  if (rootMotion)
156  {
157  var pos = Vector3.zero; // the appropriate root motion offsets are applied by timeline
158  var rot = Quaternion.identity;
159  animClip.SetCurve(string.Empty, typeof(Transform), k_PosX, SetZeroKey(curve, keyFrames, pos.x));
160  animClip.SetCurve(string.Empty, typeof(Transform), k_PosY, SetZeroKey(curve, keyFrames, pos.y));
161  animClip.SetCurve(string.Empty, typeof(Transform), k_PosZ, SetZeroKey(curve, keyFrames, pos.z));
162  animClip.SetCurve(string.Empty, typeof(Transform), k_RotX, SetZeroKey(curve, keyFrames, rot.x));
163  animClip.SetCurve(string.Empty, typeof(Transform), k_RotY, SetZeroKey(curve, keyFrames, rot.y));
164  animClip.SetCurve(string.Empty, typeof(Transform), k_RotZ, SetZeroKey(curve, keyFrames, rot.z));
165  animClip.SetCurve(string.Empty, typeof(Transform), k_RotW, SetZeroKey(curve, keyFrames, rot.w));
166  }
167 
168  return animClip;
169  }

171  public static bool IsRootMotion(EditorCurveBinding binding)
172  {
173  // Root Transform TR.
174  if (typeof(Transform).IsAssignableFrom(binding.type) && string.IsNullOrEmpty(binding.path))
175  {
176  return binding.propertyName.StartsWith(k_Pos) || binding.propertyName.StartsWith(k_Rot);
177  }
178 
179  // MotionCurves/RootCurves.
180  if (binding.type == typeof(Animator))
181  {
182  return binding.propertyName.StartsWith(k_MotionT) || binding.propertyName.StartsWith(k_MotionQ) ||
183  binding.propertyName.StartsWith(k_RootT) || binding.propertyName.StartsWith(k_RootQ);
184  }
185 
186  return false;
187  }
