228  static InputEvent InputEventType(Rect rect, int id)
229  {
230  var evt = Event.current;
231  switch (evt.GetTypeForControl(id))
232  {
233  case EventType.MouseDown:
234  if (rect.Contains(evt.mousePosition) && evt.button == 0)
235  {
236  return InputEvent.DragEnter;
237  }
238  break;
239  case EventType.MouseUp:
240  if (GUIUtility.hotControl == id)
241  {
242  return InputEvent.DragExit;
243  }
244  break;
245  case EventType.MouseDrag:
246  if (GUIUtility.hotControl == id)
247  {
248  return InputEvent.Drag;
249  }
250  break;
251  case EventType.KeyDown:
252  if (GUIUtility.hotControl == id && evt.keyCode == KeyCode.Escape)
253  {
254  return InputEvent.DragExit;
255  }
256  break;
257  }
258  return InputEvent.None;
259  }
