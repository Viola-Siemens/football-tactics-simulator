418  string ProjectText(
419  Assembly assembly,
420  Dictionary<string, string> allAssetsProjectParts,
421  List<ResponseFileData> responseFilesData,
422  HashSet<string> assemblyNames,
423  string[] roslynAnalyzerDllPaths)
424  {
425  var projectBuilder = new StringBuilder();
426  ProjectHeader(assembly, responseFilesData, roslynAnalyzerDllPaths, projectBuilder);
427  var references = new List<string>();
428 
429  foreach (string file in assembly.sourceFiles)
430  {
431  if (!HasValidExtension(file))
432  continue;
433 
434  var extension = Path.GetExtension(file).ToLower();
435  var fullFile = EscapedRelativePathFor(file);
436  if (".dll" != extension)
437  {
438  projectBuilder.Append(" <Compile Include=\"").Append(fullFile).Append("\" />").Append(k_WindowsNewline);
439  }
440  else
441  {
442  references.Add(fullFile);
443  }
444  }
445 
446  // Append additional non-script files that should be included in project generation.
447  if (allAssetsProjectParts.TryGetValue(assembly.name, out var additionalAssetsForProject))
448  projectBuilder.Append(additionalAssetsForProject);
449 
450  var responseRefs = responseFilesData.SelectMany(x => x.FullPathReferences.Select(r => r));
451  var internalAssemblyReferences = assembly.assemblyReferences
452  .Where(i => !i.sourceFiles.Any(ShouldFileBePartOfSolution)).Select(i => i.outputPath);
453  var allReferences =
454  assembly.compiledAssemblyReferences
455  .Union(responseRefs)
456  .Union(references)
457  .Union(internalAssemblyReferences)
458  .Except(roslynAnalyzerDllPaths);
459 
460  foreach (var reference in allReferences)
461  {
462  string fullReference = Path.IsPathRooted(reference) ? reference : Path.Combine(ProjectDirectory, reference);
463  AppendReference(fullReference, projectBuilder);
464  }
465 
466  if (0 < assembly.assemblyReferences.Length)
467  {
468  projectBuilder.Append(" </ItemGroup>").Append(k_WindowsNewline);
469  projectBuilder.Append(" <ItemGroup>").Append(k_WindowsNewline);
470  foreach (Assembly reference in assembly.assemblyReferences.Where(i => i.sourceFiles.Any(ShouldFileBePartOfSolution)))
471  {
472  var referencedProject = reference.outputPath;
473 
474  projectBuilder.Append(" <ProjectReference Include=\"").Append(reference.name).Append(GetProjectExtension()).Append("\">").Append(k_WindowsNewline);
475  projectBuilder.Append(" <Project>{").Append(ProjectGuid(reference.name)).Append("}</Project>").Append(k_WindowsNewline);
476  projectBuilder.Append(" <Name>").Append(reference.name).Append("</Name>").Append(k_WindowsNewline);
477  projectBuilder.Append(" <ReferenceOutputAssembly>false</ReferenceOutputAssembly>").Append(k_WindowsNewline);
478  projectBuilder.Append(" </ProjectReference>").Append(k_WindowsNewline);
479  }
480  }
481 
482  projectBuilder.Append(ProjectFooter());
483  return projectBuilder.ToString();
484  }

530  private static ILookup<string, string> GetOtherArgumentsFromResponseFilesData(List<ResponseFileData> responseFilesData)
531  {
532  var paths = responseFilesData.SelectMany(x =>
533  {
534  return x.OtherArguments.Where(a => a.StartsWith("/") || a.StartsWith("-"))
535  .Select(b =>
536  {
537  var index = b.IndexOf(":", StringComparison.Ordinal);
538  if (index > 0 && b.Length > index)
539  {
540  var key = b.Substring(1, index - 1);
541  return new KeyValuePair<string, string>(key, b.Substring(index + 1));
542  }
543 
544  const string warnaserror = "warnaserror";
545  if (b.Substring(1).StartsWith(warnaserror))
546  {
547  return new KeyValuePair<string, string>(warnaserror, b.Substring(warnaserror.Length + 1));
548  }
549 
550  return default;
551  });
552  })
553  .Distinct()
554  .ToLookup(o => o.Key, pair => pair.Value);
555  return paths;
556  }
