208  // Helper that finds the animation clip we are recording and the relative time to that clip
209  static bool GetClipAndRelativeTime(UnityEngine.Object target, WindowState state,
210  out AnimationClip outClip, out double keyTime, out bool keyInRange)
211  {
212  const float floatToDoubleError = 0.00001f;
213  outClip = null;
214  keyTime = 0;
215  keyInRange = false;
216 
217  double startTime = 0;
218  double timeScale = 1;
219  AnimationClip clip = null;
220 
221  IPlayableAsset playableAsset = target as IPlayableAsset;
222  Component component = target as Component;
223 
224  // Handle recordable playable assets
225  if (playableAsset != null)
226  {
227  var curvesOwner = AnimatedParameterUtility.ToCurvesOwner(playableAsset, state.editSequence.asset);
228  if (curvesOwner != null && state.IsTrackRecordable(curvesOwner.targetTrack))
229  {
230  if (curvesOwner.curves == null)
231  curvesOwner.CreateCurves(curvesOwner.GetUniqueRecordedClipName());
232 
233  clip = curvesOwner.curves;
234 
235  var timelineClip = curvesOwner as TimelineClip;
236  if (timelineClip != null)
237  {
238  startTime = timelineClip.start;
239  timeScale = timelineClip.timeScale;
240  }
241  }
242  }
243  // Handle recording components, including infinite clip
244  else if (component != null)
245  {
246  var asset = GetTrackForGameObject(component.gameObject, state);
247  if (asset != null)
248  {
249  clip = GetRecordingClip(asset, state, out startTime, out timeScale);
250  }
251  }
252 
253  if (clip == null)
254  return false;
255 
256  keyTime = (state.editSequence.time - startTime) * timeScale;
257  outClip = clip;
258  keyInRange = keyTime >= 0 && keyTime <= (clip.length * timeScale + floatToDoubleError);
259 
260  return true;
261  }

354  static HashSet<double> GetKeyTimes(UnityEngine.Object target, IEnumerable<PropertyModification> modifications, WindowState state)
355  {
356  var keyTimes = new HashSet<double>();
357 
358  AnimationClip animationClip;
359  double keyTime;
360  bool inRange;
361  GetClipAndRelativeTime(target, state, out animationClip, out keyTime, out inRange);
362  if (animationClip == null)
363  return keyTimes;
364 
365  var component = target as Component;
366  var playableAsset = target as IPlayableAsset;
367  var info = AnimationClipCurveCache.Instance.GetCurveInfo(animationClip);
368 
369  TimelineClip clip = null;
370  if (component != null)
371  {
372  GetTrackForGameObject(component.gameObject, state).FindRecordingClipAtTime(state.editSequence.time, out clip);
373  }
374  else if (playableAsset != null)
375  {
376  clip = FindClipWithAsset(state.editSequence.asset, playableAsset);
377  }
378 
379  foreach (var mod in modifications)
380  {
381  EditorCurveBinding temp;
382  if (HasBinding(target, mod, animationClip, out temp))
383  {
384  IEnumerable<double> keys = new HashSet<double>();
385  if (temp.isPPtrCurve)
386  {
387  var curve = info.GetObjectCurveForBinding(temp);
388  if (curve != null)
389  {
390  keys = curve.Select(x => (double)x.time);
391  }
392  }
393  else
394  {
395  var curve = info.GetCurveForBinding(temp);
396  if (curve != null)
397  {
398  keys = curve.keys.Select(x => (double)x.time);
399  }
400  }
401 
402  // Transform the times in to 'global' space using the clip
403  if (clip != null)
404  {
405  foreach (var k in keys)
406  {
407  var time = clip.FromLocalTimeUnbound(k);
408  const double eps = 1e-5;
409  if (time >= clip.start - eps && time <= clip.end + eps)
410  {
411  keyTimes.Add(time);
412  }
413  }
414  }
415  // infinite clip mode, global == local space
416  else
417  {
418  keyTimes.UnionWith(keys);
419  }
420  }
421  }
422 
423  return keyTimes;
424  }
