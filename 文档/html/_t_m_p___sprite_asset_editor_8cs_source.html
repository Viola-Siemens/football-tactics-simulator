617  void DisplayPageNavigation(ref int currentPage, int arraySize, int itemsPerPage)
618  {
619  Rect pagePos = EditorGUILayout.GetControlRect(false, 20);
620  pagePos.width /= 3;
621 
622  int shiftMultiplier = Event.current.shift ? 10 : 1; // Page + Shift goes 10 page forward
623 
624  // Previous Page
625  GUI.enabled = currentPage > 0;
626 
627  if (GUI.Button(pagePos, "Previous Page"))
628  {
629  currentPage -= 1 * shiftMultiplier;
630  //m_isNewPage = true;
631  }
632 
633  // Page Counter
634  GUI.enabled = true;
635  pagePos.x += pagePos.width;
636  int totalPages = (int)(arraySize / (float)itemsPerPage + 0.999f);
637  GUI.Label(pagePos, "Page " + (currentPage + 1) + " / " + totalPages, TMP_UIStyleManager.centeredLabel);
638 
639  // Next Page
640  pagePos.x += pagePos.width;
641  GUI.enabled = itemsPerPage * (currentPage + 1) < arraySize;
642 
643  if (GUI.Button(pagePos, "Next Page"))
644  {
645  currentPage += 1 * shiftMultiplier;
646  //m_isNewPage = true;
647  }
648 
649  // Clamp page range
650  currentPage = Mathf.Clamp(currentPage, 0, arraySize / itemsPerPage);
