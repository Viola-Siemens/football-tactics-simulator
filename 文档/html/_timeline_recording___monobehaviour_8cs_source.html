59  {

66

69

71  {

73  s_ModsToProcess.RemoveAt(s_ModsToProcess.Count - 1);
74

93

95  var toProcess = GatherRelatedModifications(modification, s_ModsToProcess);
96

99

101  // Case 900624
102  UpdatePreviewMode(toProcess, modifiedGO);

105  AddTrackOffset(animTrack, toProcess, clip, animator);

654  internal static UndoPropertyModification[] HandleEulerModifications(AnimationTrack track, TimelineClip clip, AnimationClip animClip, float time, UndoPropertyModification[] mods)
655  {
656  if (mods.Any(x => x.currentValue.propertyPath.StartsWith(kLocalEulerHint) || x.currentValue.propertyPath.StartsWith(kLocalRotation)))
657  {
658  // if there is a rotational offsets, we need to strip the euler hints, since they are used by the animation recording system
659  // over the quaternion.
660  var localToTrack = GetLocalToTrack(track, clip);
661  if (localToTrack.rotation != Quaternion.identity)
662  {
663  if (s_LastTrackWarning != track)
664  {
665  s_LastTrackWarning = track;
666  Debug.LogWarning(kRotationWarning);
667  }
668 
669  Transform transform = mods[0].currentValue.target as Transform;
670  if (transform != null)
671  {
672  var trackToLocal = TimelineAnimationUtilities.RigidTransform.Inverse(localToTrack);
673  // since the euler angles are going to be transformed, we do a best guess at a euler that gives the shortest path
674  var quatMods = mods.Where(x => !x.currentValue.propertyPath.StartsWith(kLocalEulerHint));
675  var eulerMods = FindBestEulerHint(trackToLocal.rotation * transform.localRotation, animClip, time, transform);
676  return quatMods.Union(eulerMods).ToArray();
677  }
678  return mods.Where(x => !x.currentValue.propertyPath.StartsWith(kLocalEulerHint)).ToArray();
679  }
680  }
681  return mods;

684  internal static IEnumerable<UndoPropertyModification> FindBestEulerHint(Quaternion rotation, AnimationClip clip, float time, Transform transform)
685  {
686  Vector3 euler = rotation.eulerAngles;
687 
688  var xCurve = AnimationUtility.GetEditorCurve(clip, EditorCurveBinding.FloatCurve(string.Empty, typeof(Transform), "localEulerAnglesRaw.x"));

690  var zCurve = AnimationUtility.GetEditorCurve(clip, EditorCurveBinding.FloatCurve(string.Empty, typeof(Transform), "localEulerAnglesRaw.z"));
691 
692  if (xCurve != null)
693  euler.x = xCurve.Evaluate(time);
694  if (yCurve != null)
695  euler.y = yCurve.Evaluate(time);
696  if (zCurve != null)
697  euler.z = zCurve.Evaluate(time);

699  euler = QuaternionCurveTangentCalculation.GetEulerFromQuaternion(rotation, euler);

701  return new[]

704  PropertyModificationToUndoPropertyModification(new PropertyModification {target = transform, propertyPath = kLocalEulerHint + ".y", value = euler.y.ToString() }),
