130  public static double ParseTimeCode(string timeCode, double frameRate, double defaultValue)
131  {
132  timeCode = RemoveChar(timeCode, c => char.IsWhiteSpace(c));
133  string[] sections = timeCode.Split(':');
134  if (sections.Length == 0 || sections.Length > 4)
135  return defaultValue;
136 
137  int hours = 0;
138  int minutes = 0;
139  double seconds = 0;
140  double frames = 0;
141 
142  try
143  {
144  // depending on the format of the last numbers
145  // seconds format
146  string lastSection = sections[sections.Length - 1];
147  if (Regex.Match(lastSection, @"^\d+\.\d+$").Success)
148  {
149  seconds = double.Parse(lastSection);
150  if (sections.Length > 3) return defaultValue;
151  if (sections.Length > 1) minutes = int.Parse(sections[sections.Length - 2]);
152  if (sections.Length > 2) hours = int.Parse(sections[sections.Length - 3]);
153  }
154  // frame formats
155  else
156  {
157  if (Regex.Match(lastSection, @"^\d+\[\.\d+\]$").Success)
158  {
159  string stripped = RemoveChar(lastSection, c => c == '[' || c == ']');
160  frames = double.Parse(stripped);
161  }
162  else if (Regex.Match(lastSection, @"^\d*$").Success)
163  {
164  frames = int.Parse(lastSection);
165  }
166  else
167  {
168  return defaultValue;
169  }
170 
171  if (sections.Length > 1) seconds = int.Parse(sections[sections.Length - 2]);
172  if (sections.Length > 2) minutes = int.Parse(sections[sections.Length - 3]);
173  if (sections.Length > 3) hours = int.Parse(sections[sections.Length - 4]);
174  }
175  }
176  catch (FormatException)

179  }
