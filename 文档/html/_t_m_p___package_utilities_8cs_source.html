253  private IEnumerator ScanProjectFiles()
254  {
255  m_IsAlreadyScanningProject = true;
256  string packageFullPath = EditorUtilities.TMP_EditorUtility.packageFullPath;
257 
258  // List containing assets that have been modified.
259  m_ProjectScanResults = k_ProjectScanReportDefaultText;
260  m_ModifiedAssetList.Clear();
261  m_ProgressPercentage = 0;
262 
263  // Read Conversion Data from Json file.
264  if (m_ConversionData == null)
265  m_ConversionData = JsonUtility.FromJson<AssetConversionData>(File.ReadAllText(packageFullPath + "/PackageConversionData.json"));
266 
267  // Get list of GUIDs for assets that might contain references to previous GUIDs that require updating.
268  string searchFolder = string.IsNullOrEmpty(m_ProjectFolderToScan) ? "Assets" : ("Assets/" + m_ProjectFolderToScan);
269  string[] guids = AssetDatabase.FindAssets("t:Object", new string[] { searchFolder }).Distinct().ToArray();
270 
271  k_ProjectScanLabelPrefix = "<b>Phase 1 - Filtering:</b> ";
272  m_ScanningTotalFiles = guids.Length;
273  m_ScanningCurrentFileIndex = 0;
274 
275  List<AssetFileRecord> projectFilesToScan = new List<AssetFileRecord>();
276 
277  foreach (var guid in guids)
278  {
279  if (m_CancelScanProcess)
280  break;
281 
282  string assetFilePath = AssetDatabase.GUIDToAssetPath(guid);
283 
284  m_ScanningCurrentFileIndex += 1;
285  m_ScanningCurrentFileName = assetFilePath;
286  m_ProgressPercentage = (float)m_ScanningCurrentFileIndex / m_ScanningTotalFiles;
287 
288  // Filter out file types we have no interest in searching
289  if (ShouldIgnoreFile(assetFilePath))
290  continue;
291 
292  string assetMetaFilePath = AssetDatabase.GetTextMetaFilePathFromAssetPath(assetFilePath);
293 
294  projectFilesToScan.Add(new AssetFileRecord(assetFilePath, assetMetaFilePath));
295 
296  yield return null;
297  }
298 
299  m_RemainingFilesToScan = m_ScanningTotalFiles = projectFilesToScan.Count;
300 
301  k_ProjectScanLabelPrefix = "<b>Phase 2 - Scanning:</b> ";
302 
303  for (int i = 0; i < m_ScanningTotalFiles; i++)
304  {
305  if (m_CancelScanProcess)
306  break;
307 
308  AssetFileRecord fileRecord = projectFilesToScan[i];
309 
310  Task.Run(() =>
311  {
312  ScanProjectFileAsync(fileRecord);
313 
314  m_ScanningCurrentFileName = fileRecord.assetFilePath;
315 
316  int completedScans = m_ScanningTotalFiles - Interlocked.Decrement(ref m_RemainingFilesToScan);
317 
318  m_ScanningCurrentFileIndex = completedScans;
319  m_ProgressPercentage = (float)completedScans / m_ScanningTotalFiles;
320  });
321 
322  if (i % 64 == 0)
323  yield return new WaitForSeconds(2.0f);
324 
325  }
326 
327  while (m_RemainingFilesToScan > 0 && !m_CancelScanProcess)
328  yield return null;
329 
330  m_IsAlreadyScanningProject = false;
331  m_ScanningCurrentFileName = string.Empty;
332  }

335  static void ScanProjectFileAsync(AssetFileRecord fileRecord)
336  {
337  if (m_CancelScanProcess)
338  return;
339 
340  // Read the asset data file
341  string assetDataFile = string.Empty;
342  bool hasFileChanged = false;
343 
344  try
345  {
346  assetDataFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetFilePath);
347  }
348  catch
349  {
350  // Continue to the next asset if we can't read the current one.
351  return;
352  }
353 
354  // Read the asset meta data file
355  string assetMetaFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetMetaFilePath);
356  bool hasMetaFileChanges = false;
357 
358  foreach (AssetConversionRecord record in m_ConversionData.assetRecords)
359  {
360  if (assetDataFile.Contains(record.target))
361  {
362  hasFileChanged = true;
363 
364  assetDataFile = assetDataFile.Replace(record.target, record.replacement);
365  }
366 
368  if (assetMetaFile.Contains(record.target))
369  {
370  hasMetaFileChanges = true;
371 
372  assetMetaFile = assetMetaFile.Replace(record.target, record.replacement);
373  }
374  }
375 
376  if (hasFileChanged)
377  {
378  AssetModificationRecord modifiedAsset;
379  modifiedAsset.assetFilePath = fileRecord.assetFilePath;
380  modifiedAsset.assetDataFile = assetDataFile;
381 
382  m_ModifiedAssetList.Add(modifiedAsset);
383 
384  m_ProjectScanResults += fileRecord.assetFilePath + "\n";
385  }
386 
387  if (hasMetaFileChanges)
388  {
389  AssetModificationRecord modifiedAsset;
390  modifiedAsset.assetFilePath = fileRecord.assetMetaFilePath;
391  modifiedAsset.assetDataFile = assetMetaFile;
392 
393  m_ModifiedAssetList.Add(modifiedAsset);
394 
395  m_ProjectScanResults += fileRecord.assetMetaFilePath + "\n";
396  }
397  }
