232  private bool MoveNextAndUpdateYieldObject()
233  {
234  var result = m_RunStep.MoveNext();
235 
236  if (result)
237  {
238  m_CurrentYieldObject = m_RunStep.Current;
239  while (m_CurrentYieldObject is IEnumerator) // going deeper
240  {
241  var currentEnumerator = (IEnumerator)m_CurrentYieldObject;
242 
243  // go deeper and add parent to stack
244  StepStack.Push(m_RunStep);
245 
246  m_RunStep = currentEnumerator;
247  m_CurrentYieldObject = m_RunStep.Current;
248  }
249 
250  if (StepStack.Count > 0 && m_CurrentYieldObject != null) // not null and not IEnumerator, nested
251  {
252  Debug.LogError("EditMode test can only yield null, but not <" + m_CurrentYieldObject.GetType().Name + ">");
253  }
254 
255  return true;
256  }
257 
258  if (StepStack.Count == 0) // done
259  return false;
260 
261  m_RunStep = StepStack.Pop(); // going up
262  return MoveNextAndUpdateYieldObject();
263  }
