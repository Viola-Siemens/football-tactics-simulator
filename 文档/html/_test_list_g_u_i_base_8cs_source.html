407  private TestRunnerFilter[] GetSelectedTestsAsFilter(IEnumerable<int> selectedIDs)
408  {
409  var namesToRun = new List<string>();
410  var assembliesForNamesToRun = new List<string>();
411  var exactNamesToRun = new List<string>();
412  var assembliesToRun = new List<string>();
413  foreach (var lineId in selectedIDs)
414  {
415  var line = m_TestListTree.FindItem(lineId);
416  if (line is TestTreeViewItem)
417  {
418  var testLine = line as TestTreeViewItem;
419  if (testLine.IsGroupNode && !testLine.FullName.Contains("+"))
420  {
421  if (testLine.parent != null && testLine.parent.displayName == "Invisible Root Item")
422  {
423  //Root node selected. Use an empty TestRunnerFilter to run every test
424  return new[] {new TestRunnerFilter()};
425  }
426 
427  if (testLine.FullName.EndsWith(".dll", StringComparison.OrdinalIgnoreCase))
428  {
429  assembliesToRun.Add(TestRunnerFilter.AssemblyNameFromPath(testLine.FullName));
430  }
431  else
432  {
433  namesToRun.Add($"^{Regex.Escape(testLine.FullName)}$");
434  var assembly = TestRunnerFilter.AssemblyNameFromPath(testLine.GetAssemblyName());
435  if (!string.IsNullOrEmpty(assembly) && !assembliesForNamesToRun.Contains(assembly))
436  {
437  assembliesForNamesToRun.Add(assembly);
438  }
439  }
440  }
441  else
442  {
443  exactNamesToRun.Add(testLine.FullName);
444  }
445  }
446  }
447 
448  var filters = new List<TestRunnerFilter>();
449 
450  if (assembliesToRun.Count > 0)
451  {
452  filters.Add(new TestRunnerFilter()
453  {
454  assemblyNames = assembliesToRun.ToArray()
455  });
456  }
457 
458  if (namesToRun.Count > 0)
459  {
460  filters.Add(new TestRunnerFilter()
461  {
462  groupNames = namesToRun.ToArray(),
463  assemblyNames = assembliesForNamesToRun.ToArray()
464  });
465  }
466 
467  if (exactNamesToRun.Count > 0)
468  {
469  filters.Add(new TestRunnerFilter()
470  {
471  testNames = exactNamesToRun.ToArray()
472  });
473  }
474 
475  if (filters.Count == 0)
476  {
477  filters.Add(new TestRunnerFilter());
478  }
479 
480  var categories = m_TestRunnerUIFilter.CategoryFilter.ToArray();
481  if (categories.Length > 0)
482  {
483  foreach (var filter in filters)
484  {
485  filter.categoryNames = categories;
486  }
487  }
488 
489  return filters.ToArray();
490  }
